function getOrCreateFolder(parentFolder, folderName) {
  var folders = parentFolder.getFoldersByName(folderName);
  if (folders.hasNext()) {
    return folders.next();
  } else {
    return parentFolder.createFolder(folderName);
  }
}

function createFolderNameFromSubject(subject) {
  // 提取主旨中的 [From: ***] 部分
  var match = subject.match(/\[From:\s*(.*?)\]/i);
  var folderName = match ? match[1].trim() : "未命名"; // 如果未找到，則使用"未命名"
  
  // 清理資料夾名稱中的非法字元，只保留字母、數字、空格、破折號和 @ 符號
  folderName = folderName.replace(/[^\w\s@-]/gi, ''); // 僅保留字母、數字、空格、@ 和破折號
  
  return folderName;
}

function saveEmailAsGoogleDocWithAttachments(message, folder) {
  var subject = message.getSubject().replace(/[^\w\s@-]/gi, '');
  var date = Utilities.formatDate(message.getDate(), Session.getScriptTimeZone(), 'yyyyMMdd-HHmmss');
  var docFileName = subject + " - " + date;
  
  var htmlBody = message.getBody(); // 獲取郵件的HTML內容
  var attachments = message.getAttachments();
  
  // 創建Google Docs文件
  var doc = DocumentApp.create(docFileName);
  var body = doc.getBody();
  
  // 將HTML內容以純文字的方式插入到Google Docs文件中
  body.appendParagraph("郵件內容:");
  body.appendParagraph(htmlBody.replace(/<[^>]+>/g, '')); // 去除HTML標籤
  
  // 處理圖片附件，保存到資料夾中，並插入到Google Docs文件中
  for (var i = 0; i < attachments.length; i++) {
    var attachment = attachments[i];
    if (attachment.getContentType().indexOf("image") !== -1) {
      var imageFile = folder.createFile(attachment);
      body.appendParagraph("圖片: " + attachment.getName());
      body.appendImage(imageFile.getBlob());
    } else {
      // 保存非圖片附件，如履歷
      folder.createFile(attachment);
      body.appendParagraph("附件: " + attachment.getName());
    }
  }
  
  // 將Google Docs文件移動到指定資料夾
  var docFile = DriveApp.getFileById(doc.getId());
  folder.addFile(docFile);
  DriveApp.getRootFolder().removeFile(docFile);
}
